                    </Button>
                  </Link>
                </div>
              </div>
            )}

            <div className="mt-6 pt-6 border-t">
              <h3 className="text-lg font-montserrat font-semibold mb-3">
                ¬øNecesita ayuda?
              </h3>
              <p className="text-sm text-gray-600 mb-4">
                Si tiene dudas sobre esta pieza o necesita asesoramiento t√©cnico, 
                contacte con nuestro equipo especializado.
              </p>
              <Link href="/contacto">
                <Button className="w-full">
                  Formulario de contacto
                </Button>
              </Link>
            </div>
          </CardContent>
        </Card>
      </div>

      {/* Secci√≥n de piezas relacionadas */}
      {relatedPartsData?.parts && relatedPartsData.parts.length > 0 && (
        <div className="mt-16">
          <div className="flex items-center justify-between mb-6">
            <h2 className="text-2xl font-montserrat font-bold">
              Otras piezas disponibles para {data?.vehicles?.[0] ? data.vehicles[0].descripcion : 'este veh√≠culo'}
            </h2>
            {relatedPartsData.vehicleId && (
              <Link href={`/vehiculos/${relatedPartsData.vehicleId}`}>
                <Button variant="outline">
                  Ver todas las piezas
                  <ArrowRight className="ml-2 h-4 w-4" />
                </Button>
              </Link>
            )}
          </div>

          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
            {relatedPartsData.parts.slice(0, 8).map((relatedPart) => (
              <PartCard 
                key={relatedPart.id} 
                part={relatedPart} 
                className="h-full"
                vehicleContextId={data?.vehicles?.[0]?.id}
              />
              ))}
            </div>
        </div>
      )}
    </div>
  );
};
